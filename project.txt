def takecommand():
    r=sr.Recognizer()

        print("Recognizing...") 
        query=r.recognize_google(audio,language='en-in')
        print("user said", query)
        



    except Exception as e:
        print(e)
        speak("say that again please")
        query="nothing"
    return query
